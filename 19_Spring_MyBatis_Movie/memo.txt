drop table movie;
create table movie(
	num number not null  primary key,
	title varchar2(20) not null,
	nation varchar2(25) not null,
	genre varchar2(40),
	grade varchar2(5),
	actor varchar2(50)
);

drop sequence mv_seq;
create sequence mv_seq
	increment by 1
	start with 1 
	minvalue 1
	maxvalue 10000
	nocache;

insert into movie(num,title,nation,genre,grade,actor)
values(mv_seq.nextval,'외계인','한국','액션,판타지','15','류준열,김우빈,김태리');

commit;
